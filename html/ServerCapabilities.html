0011    public struct SaveOptions: Codable {
0012        public let includeText: Bool?
0013    }

0015    public struct TextDocumentSyncOptions: Codable {
0016        public let openClose: Bool?
0017        public let change: TextDocumentSyncKind?
0018        public let willSave: Bool?
0019        public let willSaveWaitUntil: Bool?
0020        public let save: SaveOptions?
0021    
0022        public var effectiveSave: SaveOptions {
0023            return save ?? SaveOptions(includeText: false)
0024        }
0025    }
0026    
0027    public enum ServerCapabilitiesTextDocumentSync: Codable {
0028        case options(TextDocumentSyncOptions)
0029        case kind(TextDocumentSyncKind)
0030    
0031        public init(from decoder: Decoder) throws {
0032            let container = try decoder.singleValueContainer()
0033    
0034            do {
0035                let value = try container.decode(TextDocumentSyncOptions.self)
0036                self = .options(value)
0037            } catch DecodingError.typeMismatch {
0038                let value = try container.decode(TextDocumentSyncKind.self)
0039                self = .kind(value)
0040            }
0041        }
0042    
0043        public func encode(to encoder: Encoder) throws {
0044            var container = encoder.singleValueContainer()
0045    
0046            switch self {
0047            case .options(let value):
0048                try container.encode(value)
0049            case .kind(let value):
0050                try container.encode(value)
0051            }
0052        }
0053    
0054        public var effectiveOptions: TextDocumentSyncOptions {
0055            switch self {
0056            case .options(let value):
0057                return value
0058            case .kind(let value):
0059                return TextDocumentSyncOptions(openClose: false, change: value, willSave: false, willSaveWaitUntil: false, save: nil)
0060            }
0061        }
0062    }

0077    public enum ServerCapabilityTypeDefinitionProvider: Codable {
0078        case boolean(Bool)
0079        case registrationOptions(TextDocumentAndStaticRegistrationOptions)
0080    
0081        public init(from decoder: Decoder) throws {
0082            let container = try decoder.singleValueContainer()
0083    
0084            do {
0085                let value = try container.decode(Bool.self)
0086                self = .boolean(value)
0087            } catch DecodingError.typeMismatch {
0088                let value = try container.decode(TextDocumentAndStaticRegistrationOptions.self)
0089                self = .registrationOptions(value)
0090            }
0091        }
0092    
0093        public func encode(to encoder: Encoder) throws {
0094            var container = encoder.singleValueContainer()
0095    
0096            switch self {
0097            case .boolean(let value):
0098                try container.encode(value)
0099            case .registrationOptions(let value):
0100                try container.encode(value)
0101            }
0102        }
0103    }
