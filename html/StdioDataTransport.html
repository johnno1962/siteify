0009    import Foundation
0010    
0011    public class StdioDataTransport: DataTransport {
0012        public let stdinPipe: Pipe
0013        public let stdoutPipe: Pipe
0014        public let stderrPipe: Pipe
0015        var readHandler: ReadHandler?
0016        private var closed: Bool
0017        
0018        public init() {
0019            self.stdinPipe = Pipe()
0020            self.stdoutPipe = Pipe()
0021            self.stderrPipe = Pipe()
0022            self.readHandler = nil
0023            self.closed = false
0024            
0025            stdoutPipe.fileHandleForReading.readabilityHandler = { [unowned self] handle in
0026                let data = handle.availableData
0027    
0028                guard data.count > 0 else {
0029                    return
0030                }
0031                
0032                self.readHandler?(data)
0033            }
0034            
0035            stderrPipe.fileHandleForReading.readabilityHandler = { handle in
0036                let data = handle.availableData
0037                
0038                guard data.count > 0 else {
0039                    return
0040                }
0041                
0042                if let string = String(bytes: data, encoding: .utf8) {
0043                    print("stderr: \(string)")
0044                }
0045            }
0046        }
0047        
0048        public func write(_ data: Data) {
0049            if closed {
0050                return
0051            }
0052    
0053            stdinPipe.fileHandleForWriting.write(data)
0054        }
0055        
0056        public func setReaderHandler(_ handler: @escaping (Data) -> Void) {
0057            self.readHandler = handler
0058        }
0059    
0060        public func close() {
0061            if closed {
0062                return
0063            }
0064            
0065            closed = true
0066    
0067            [stdoutPipe, stderrPipe, stdinPipe].forEach { (pipe) in
0068                pipe.fileHandleForWriting.closeFile()
0069                pipe.fileHandleForReading.closeFile()
0070            }
0071        }
0072    }
