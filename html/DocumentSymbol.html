0009    import Foundation
0010    
0011    public struct DocumentSymbolParams {
0012        public let textDocument: TextDocumentIdentifier
0013    
0014        public init(textDocument: TextDocumentIdentifier) {
0015            self.textDocument = textDocument
0016        }
0017    }
0018    
0019    extension DocumentSymbolParams: Codable {
0020    }
0021    
0022    extension DocumentSymbolParams: Equatable {
0023    }
0024    
0025    public enum DocumentSymbolResponse: Codable {
0026        case documentSymbols([DocumentSymbol])
0027        case symbolInformation([SymbolInformation])
0028    
0029        public init(from decoder: Decoder) throws {
0030            let container = try decoder.singleValueContainer()
0031    
0032            do {
0033                let value = try container.decode([DocumentSymbol].self)
0034                self = .documentSymbols(value)
0035            } catch is DecodingError {
0036                let value = try container.decode([SymbolInformation].self)
0037                self = .symbolInformation(value)
0038            }
0039        }
0040    
0041        public func encode(to encoder: Encoder) throws {
0042            var container = encoder.singleValueContainer()
0043    
0044            switch self {
0045            case .documentSymbols(let value):
0046                try container.encode(value)
0047            case .symbolInformation(let value):
0048                try container.encode(value)
0049            }
0050        }
0051    }
0052    
0053    extension DocumentSymbolResponse: Equatable {
0054    }
