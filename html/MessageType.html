0009    import Foundation
0010    
0011    public enum MessageType: Int, Codable {
0012        case error = 1
0013        case warning = 2
0014        case info = 3
0015        case log = 4
0016    }
0017    
0018    extension MessageType: CustomStringConvertible {
0019        public var description: String {
0020            switch self {
0021            case .error:
0022                return "error"
0023            case .warning:
0024                return "warning"
0025            case .info:
0026                return "info"
0027            case .log:
0028                return "log"
0029            }
0030        }
0031    }
