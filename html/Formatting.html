0011    public struct FormattingOptions: Codable {
0012        public let tabSize: Int
0013        public let insertSpaces: Bool
0014    
0015        public init(tabSize: Int, insertSpaces: Bool) {
0016            self.tabSize = tabSize
0017            self.insertSpaces = insertSpaces
0018        }
0019    }
