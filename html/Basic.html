0052    public enum LanguageIdentifier: String, Codable, CaseIterable {
0053        case go = "go"
0054        case json = "json"
0055        case swift = "swift"
0056        case c = "c"
0057        case cpp = "cpp"
0058        case objc = "objective-c"
0059        case objcpp = "objective-cpp"
0060    
0061        static let fileExtensions = [
0062            "go": .go,
0063            "json": .json,
0064            "swift": .swift,
0065            "c": c,
0066            "C": .cpp,
0067            "cpp": .cpp,
0068            "m": .objc,
0069            "mm": .objcpp,
0070        ]
0071    
0072        public struct UnsupportedFileType: Error {
0073            let msg: String
0074        }
0075    
0076        public init(for url: URL) throws {
0077            let ext = url.pathExtension
0078            guard let languageID =
0079                LanguageIdentifier.fileExtensions[ext] else {
0080                throw UnsupportedFileType(msg: "Unsupported file extension \(ext)")
0081            }
0082            self = languageID
0083        }
0084    }

0086    public struct TextDocumentItem: Codable {
0087        public let uri: DocumentUri
0088        public let languageId: LanguageIdentifier
0089        public let version: Int
0090        public let text: String
0091        
0092        public init(uri: DocumentUri, languageId: LanguageIdentifier, version: Int, text: String) {
0093            self.uri = uri
0094            self.languageId = languageId
0095            self.version = version
0096            self.text = text
0097        }
0098    
0099        public init(contentsAt path: String, version: Int = 1) throws {
0100            let url = URL(fileURLWithPath: path)
0101            self.uri = url.absoluteString
0102            self.languageId = try LanguageIdentifier(for: url)
0103            self.version = version
0104            self.text = try String(contentsOf: url)
0105        }
0106    }
