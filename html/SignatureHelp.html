0009    import Foundation
0010    
0011    public enum SignatureHelpDocumentation: Codable {
0012        case string(String)
0013        case markup(MarkupContent)
0014    
0015        public init(from decoder: Decoder) throws {
0016            let container = try decoder.singleValueContainer()
0017    
0018            if let value = try? container.decode(String.self) {
0019                self = .string(value)
0020            } else {
0021                let value = try container.decode(MarkupContent.self)
0022                self = .markup(value)
0023            }
0024        }
0025    
0026        public func encode(to encoder: Encoder) throws {
0027            var container = encoder.singleValueContainer()
0028    
0029            switch self {
0030            case .string(let value):
0031                try container.encode(value)
0032            case .markup(let value):
0033                try container.encode(value)
0034            }
0035        }
0036    }
