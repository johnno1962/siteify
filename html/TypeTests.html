0009    import XCTest
0010    @testable import SwiftLSPClient
0011    
0012    class TypeTests: XCTestCase {
0013        func testSomething() {
0014            let json = """
0015    {"id":2,"result":[{"name":"something","kind":12,"location":{"uri":"file:///hello.go","range":{"start":{"line":19,"character":5},"end":{"line":19,"character":13}}}}],"jsonrpc":"2.0"}
0016    """
0017            let data = json.data(using: .utf8)!
0018            let response = try! JSONDecoder().decode(JSONRPCResultResponse<DocumentSymbolResponse>.self, from: data)
0019    
0020            let location = Location(uri: "file:///hello.go", range: LSPRange(startPair: (19, 5), endPair: (19, 13)))
0021            let symbolInfo = SymbolInformation(name: "something", kind: 12, deprecated: nil, location: location, containerName: nil)
0022            XCTAssertEqual(response.result, DocumentSymbolResponse.symbolInformation([symbolInfo]))
0023        }
0024    }
