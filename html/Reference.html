0009    import Foundation
0010    
0011    public struct ReferenceContext: Codable {
0012        public let includeDeclaration: Bool
0013    
0014        public init(includeDeclaration: Bool = false) {
0015            self.includeDeclaration = includeDeclaration
0016        }
0017    }
0018    
0019    public struct ReferenceParams: Codable {
0020        public let textDocument: TextDocumentIdentifier
0021        public let position: Position
0022        public let context: ReferenceContext
0023    
0024        public init(textDocument: TextDocumentIdentifier, position: Position,
0025                    context: ReferenceContext) {
0026            self.textDocument = textDocument
0027            self.position = position
0028            self.context = context
0029        }
0030    
0031        public init(textDocument: TextDocumentIdentifier, position: Position, includeDeclaration: Bool = false) {
0032            let ctx = ReferenceContext(includeDeclaration: includeDeclaration)
0033    
0034            self.init(textDocument: textDocument, position: position, context: ctx)
0035        }
0036    }
0037    
0038    public typealias ReferenceResponse = [Location]
