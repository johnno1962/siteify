0009    import Foundation
0010    
0011    public func parallelise(threads: Int,
0012                            name: String = "Parallelize",
0013                            work: () -> (() -> Void)?) {
0014        let concurrentQueue = DispatchQueue(label: name, qos: .background, attributes: .concurrent)
0015        let synchroniser = DispatchSemaphore(value: threads)
0016        let threadGroup = DispatchGroup()
0017    
0018        while let nextWork = work() {
0019            synchroniser.wait()
0020            threadGroup.enter()
0021    
0022            concurrentQueue.async {
0023                nextWork()
0024                threadGroup.leave()
0025                synchroniser.signal()
0026            }
0027        }
0028    
0029        threadGroup.wait()
0030    }
