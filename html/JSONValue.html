0009    import Foundation
0010    
0011    public enum JSONValue: Codable {
0012        case null
0013        case string(String)
0014        case int(Int)
0015        case double(Double)
0016        case bool(Bool)
0017        case object([String : JSONValue])
0018        case array([JSONValue])
0019    
0020        public init(from decoder: Decoder) throws {
0021            let container = try decoder.singleValueContainer()
0022    
0023            if let value = try? container.decode(String.self) {
0024                self = .string(value)
0025            } else if let value = try? container.decode(Int.self) {
0026                self = .int(value)
0027            } else if let value = try? container.decode(Double.self) {
0028                self = .double(value)
0029            } else if let value = try? container.decode(Bool.self) {
0030                self = .bool(value)
0031            } else if let value = try? container.decode([String : JSONValue].self) {
0032                self = .object(value)
0033            } else if let value = try? container.decode([JSONValue].self) {
0034                self = .array(value)
0035            } else if container.decodeNil() {
0036                self = .null
0037            } else {
0038                let ctx = DecodingError.Context(codingPath: container.codingPath, debugDescription: "Unknown JSON Type")
0039                throw DecodingError.typeMismatch(JSONValue.self, ctx)
0040            }
0041        }
0042    
0043        public func encode(to encoder: Encoder) throws {
0044            var container = encoder.singleValueContainer()
0045    
0046            switch self {
0047            case .null:
0048                try container.encodeNil()
0049            case .string(let value):
0050                try container.encode(value)
0051            case .int(let value):
0052                try container.encode(value)
0053            case .double(let value):
0054                try container.encode(value)
0055            case .bool(let value):
0056                try container.encode(value)
0057            case .object(let value):
0058                try container.encode(value)
0059            case .array(let value):
0060                try container.encode(value)
0061            }
0062        }
0063    }
