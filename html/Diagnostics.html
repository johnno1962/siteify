0009    import Foundation
0010    
0011    public struct DiagnosticRelatedInformation: Codable {
0012        public let location: Location
0013        public let message: String
0014    }
0015    
0016    public enum DiagnosticCode: Codable {
0017        case number(Int)
0018        case string(String)
0019    
0020        public init(from decoder: Decoder) throws {
0021            let container = try decoder.singleValueContainer()
0022    
0023            if let value = try? container.decode(String.self) {
0024                self = .string(value)
0025            } else {
0026                let value = try container.decode(Int.self)
0027                self = .number(value)
0028            }
0029        }
0030    
0031        public func encode(to encoder: Encoder) throws {
0032            var container = encoder.singleValueContainer()
0033    
0034            switch self {
0035            case .string(let value):
0036                try container.encode(value)
0037            case .number(let value):
0038                try container.encode(value)
0039            }
0040        }
0041    }

0046        public let code: DiagnosticCode?

0049        public let relatedInformation: [DiagnosticRelatedInformation]?
0050    }
