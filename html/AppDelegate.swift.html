0026        @IBAction func downloadToolchain(_ sender: NSButton) {
0027            NSWorkspace.shared.open(URL(string: "https://swift.org/download/")!)
0028        }
0029    
0030        @IBAction func selectToolchain(_ sender: NSButton) {
0031            let open = NSOpenPanel()
0032            open.directoryURL = URL(fileURLWithPath: "/Library/Developer/Toolchains")
0033            open.allowsMultipleSelection = false
0034            open.canChooseDirectories = true
0035            open.canChooseFiles = true
0036            open.prompt = NSLocalizedString("Select Toochain to use",
0037                                            tableName: "Toochain",
0038                                            comment: "Toochain")
0039            if open.runModal() == .OK {
0040                toolchainPath = open.url?.path
0041            }
0042        }
0043    
0044        @IBAction func selectProject(_ sender: NSButton) {
0045            let open = NSOpenPanel()
0046            open.allowsMultipleSelection = false
0047            open.canChooseDirectories = true
0048            open.prompt = NSLocalizedString("Select project to convert",
0049                                            tableName: "Project Directory",
0050                                            comment: "Project Directory")
0051            if open.runModal() == .OK {
0052                projectRoot = open.url?.path
0053            }
0054        }
0055    
0056        @IBAction func convertProject(_ sender: NSButton) {
0057            let converter = MySiteify(toolchainPath: toolchainPath!, projectRoot: projectRoot!)
0058            converter.generateSite(into: projectRoot!+"/html")
0059        }
0060    }
