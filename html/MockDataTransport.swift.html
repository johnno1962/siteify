0009    import Foundation
0010    import SwiftLSPClient
0011    
0012    class MockDataTransport: DataTransport {
0013        var writtenData: [Data]
0014        var readHandler: ReadHandler?
0015    
0016        init() {
0017            self.writtenData = []
0018            self.readHandler = nil
0019        }
0020    
0021        func write(_ data: Data) {
0022            writtenData.append(data)
0023        }
0024    
0025        func setReaderHandler(_ handler: @escaping ReadHandler) {
0026            readHandler = handler
0027        }
0028    
0029        func close() {
0030        }
0031    
0032        func mockRead(_ data: Data) {
0033            self.readHandler?(data)
0034        }
0035    
0036        func mockRead(_ string: String) {
0037            mockRead(string.data(using: .utf8)!)
0038        }
0039    }
